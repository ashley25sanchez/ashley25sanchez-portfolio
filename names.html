# create the dict and save it to a file d={ 'part1':[ 'Bloody', 'Gruesome', 'Evil', 'King',], 'part2':[ 'Buttocks', 'Elf', 'Dimple', 'Shoo',], } import pickle f=open('syllables','w') pickle.dump(d,f) f.close() # read the dict back in from the file f1=open('syllables','r') sd=pickle.load(f1) f1.close() import random first_part=sd['part1'][random.randint(0,len(sd['part1'])-1)] second_part=sd['part2'][random.randint(0,len(sd['part2'])-1)] print '%s%s'%(first_part,second_part)
